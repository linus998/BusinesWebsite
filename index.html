Bouncing Ball Simulation

A Python simulation that demonstrates physics-based motion with gravity for a bouncing ball. And also a small piece of the code! 

import pygame

def simulate_ball():
    pygame.init()
    screen = pygame.display.set_mode((500, 500))
    ball = pygame.Rect(250, 250, 20, 20)
    velocity = [2, -5]
    gravity = 0.5

    while True:
        for event in pygame.event.get():
            if event.type == pygame.QUIT:
                return
        velocity[1] += gravity
        ball = ball.move(velocity)
        if ball.bottom >= 500 or ball.top <= 0:
            velocity[1] = -velocity[1] * 0.8
        pygame.display.flip()

Data Scraper for TikTok

An asynchronous data scraper built using TikTokAPIpy to collect video metadata efficiently. A very minimalistic piece of the code to show.

from tiktokapipy.async_api import AsyncTikTokAPI
import asyncio

async def scrape_tiktok():
    async with AsyncTikTokAPI() as api:
        video_data = await api.video("video_url")
        print(video_data)

asyncio.run(scrape_tiktok())
